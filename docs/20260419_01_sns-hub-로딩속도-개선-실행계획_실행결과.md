```python
from __future__ import annotations

from urllib.parse import quote


META_FIELDS = [
    "sequence_id",
    "platform_id",
    "sns_platform",
    "code",
    "username",
    "display_name",
    "url",
    "canonical_url",
    "created_at",
    "date",
    "source",
    "full_text_preview",
    "full_text_length",
    "media_count",
    "local_images_count",
    "thumbnail",
    "is_detail_collected",
    "is_merged_thread",
]


def _normalize_threads_url(url: str) -> str:
    return (
        url.replace("https://www.threads.net/", "https://www.threads.com/")
        .replace("https://threads.net/", "https://www.threads.com/")
        .replace("https://threads.com/", "https://www.threads.com/")
    )


def canonicalize_url(post: dict) -> str:
    raw_url = str(post.get("url") or post.get("post_url") or post.get("source_url") or "")
    platform = str(post.get("sns_platform") or "").lower()
    username = post.get("username") or post.get("user") or ""
    code = post.get("code") or post.get("platform_id") or ""

    if "thread" in platform:
        if username and code:
            return f"https://www.threads.com/@{username}/post/{code}"
        if raw_url:
            return _normalize_threads_url(raw_url)
    return raw_url


def build_thumbnail(post: dict) -> str | None:
    local_images = post.get("local_images") or []
    media = post.get("media") or []
    if local_images:
        return local_images[0]
    if not media:
        return None
    first = media[0]
    if "licdn.com" in first:
        return first
    return f"https://wsrv.nl/?url={quote(first, safe='')}&output=webp"


def build_post_meta(post: dict) -> dict:
    full_text = str(post.get("full_text") or "")
    media = post.get("media") or []
    local_images = post.get("local_images") or []

    enriched = {
        **post,
        "canonical_url": canonicalize_url(post),
        "full_text_preview": full_text[:200],
        "full_text_length": len(full_text),
        "media_count": len(media),
        "local_images_count": len(local_images),
        "thumbnail": build_thumbnail(post),
        "_searchable": "\n".join(
            [
                full_text,
                str(post.get("display_name") or ""),
                str(post.get("username") or post.get("user") or ""),
            ]
        ).lower(),
    }
    return {field: enriched.get(field) for field in META_FIELDS}
```

```python
import gzip

from utils.post_meta import META_FIELDS, build_post_meta, canonicalize_url


_POSTS_CACHE = {"path": None, "mtime": 0, "size": 0, "posts_full": [], "posts_meta": [], "etag": None}


def _load_latest_posts():
    pattern = os.path.join(OUTPUT_TOTAL_DIR, "total_full_*.json")
    files = [
        path
        for path in glob.glob(pattern)
        if re.fullmatch(r"total_full_\d{8}\.json", os.path.basename(path))
    ]
    if not files:
        raise FileNotFoundError("Data file not found")

    latest_path = sorted(files)[-1]
    stat = os.stat(latest_path)
    if (
        _POSTS_CACHE["path"] == latest_path
        and _POSTS_CACHE["mtime"] == stat.st_mtime
        and _POSTS_CACHE["size"] == stat.st_size
    ):
        return _POSTS_CACHE["posts_full"], _POSTS_CACHE["posts_meta"], _POSTS_CACHE["etag"]

    with open(latest_path, "r", encoding="utf-8-sig") as handle:
        payload = json.load(handle)

    posts_full = []
    posts_meta = []
    for raw_post in payload.get("posts", []):
        meta = build_post_meta(raw_post)
        posts_meta.append(meta)
        posts_full.append(
            {
                **raw_post,
                **meta,
                "_searchable": "\n".join(
                    [
                        str(raw_post.get("full_text") or ""),
                        str(raw_post.get("display_name") or ""),
                        str(raw_post.get("username") or raw_post.get("user") or ""),
                    ]
                ).lower(),
            }
        )

    etag = f'"{int(stat.st_mtime)}-{stat.st_size}"'
    _POSTS_CACHE.update(
        {
            "path": latest_path,
            "mtime": stat.st_mtime,
            "size": stat.st_size,
            "posts_full": posts_full,
            "posts_meta": posts_meta,
            "etag": etag,
        }
    )
    return posts_full, posts_meta, etag


@app.after_request
def add_json_caching_headers(response):
    if response.mimetype == "application/json":
        try:
            _, _, etag = _load_latest_posts()
            response.headers["ETag"] = etag
            response.headers["Cache-Control"] = "private, max-age=0, must-revalidate"
            if request.headers.get("If-None-Match") == etag:
                response.status_code = 304
                response.set_data(b"")
                return response
            if "gzip" in request.headers.get("Accept-Encoding", "").lower() and len(response.get_data()) >= 512:
                response.set_data(gzip.compress(response.get_data()))
                response.headers["Content-Encoding"] = "gzip"
                response.headers["Vary"] = "Accept-Encoding"
        except FileNotFoundError:
            pass
    return response


@app.route("/api/posts", methods=["GET"])
def get_posts():
    _, posts_meta, _ = _load_latest_posts()
    return jsonify({"posts": posts_meta})


@app.route("/api/post/<int:sequence_id>", methods=["GET"])
def get_post_detail(sequence_id: int):
    posts_full, _, _ = _load_latest_posts()
    match = next((post for post in posts_full if int(post.get("sequence_id") or 0) == sequence_id), None)
    if not match:
        return jsonify({"error": "Post not found"}), 404
    detail = {key: value for key, value in match.items() if key != "_searchable"}
    return jsonify(detail)
```

```python
def _normalize_platform_filter(raw: str) -> str:
    value = (raw or "").strip().lower()
    if value == "x":
        return "twitter"
    return value


@app.route("/api/search", methods=["GET"])
def search_posts():
    q = (request.args.get("q") or "").strip()
    if not q:
        return jsonify({"error": "q is required"}), 400

    platform = _normalize_platform_filter(request.args.get("platform"))
    limit = min(max(int(request.args.get("limit", 500)), 1), 1000)
    offset = max(int(request.args.get("offset", 0)), 0)

    posts_full, _, _ = _load_latest_posts()
    matches = []
    for post in posts_full:
        if platform and (post.get("sns_platform") or "").lower() != platform:
            continue
        if q.lower() in post["_searchable"]:
            matches.append(build_post_meta(post))

    sliced = matches[offset : offset + limit]
    return jsonify(
        {
            "posts": sliced,
            "query": q,
            "total_matched": len(matches),
            "returned": len(sliced),
            "limit": limit,
            "offset": offset,
        }
    )


@app.route("/api/auto-tag/apply", methods=["POST"])
def apply_auto_tag_rules():
    payload = request.get_json(silent=True) or {}
    rules = payload.get("rules")
    if not isinstance(rules, list):
        return jsonify({"error": "rules is required"}), 400

    posts_full, _, _ = _load_latest_posts()
    result = {}
    matched_post_count = 0

    for post in posts_full:
        match_bucket = set()
        for rule in rules:
            keyword = str(rule.get("keyword") or "").strip().lower()
            tag = str(rule.get("tag") or "").strip()
            match_field = str(rule.get("match_field") or "all")
            if not keyword or not tag:
                continue

            haystack = str(post.get("full_text") or "")
            if match_field == "all":
                haystack = "\n".join(
                    [
                        haystack,
                        str(post.get("display_name") or ""),
                        str(post.get("username") or post.get("user") or ""),
                    ]
                )

            if keyword in haystack.lower():
                match_bucket.add(tag)

        if match_bucket:
            matched_post_count += 1
            result[post["canonical_url"]] = sorted(match_bucket)

    return jsonify(
        {
            "url_to_auto_tags": result,
            "matched_post_count": matched_post_count,
            "rule_count": len(rules),
        }
    )
```

```js
function mergeAutoTags(urlToAutoTags) {
    Object.entries(urlToAutoTags).forEach(([url, tags]) => {
        const merged = new Set(postTags[url] || []);
        tags.forEach((tag) => merged.add(tag));
        postTags[url] = [...merged];
    });
}

async function applyAutoTagRules() {
    const rules = JSON.parse(localStorage.getItem("sns_auto_tag_rules") || "[]");
    if (rules.length === 0) {
        renderPosts();
        return;
    }

    const res = await fetch("/api/auto-tag/apply", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ rules: rules.map((rule) => ({ ...rule, match_field: "all" })) }),
    });
    const { url_to_auto_tags } = await res.json();
    mergeAutoTags(url_to_auto_tags);
    localStorage.setItem("sns_tags", JSON.stringify(postTags));
    await fetch("/api/save-tags", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(postTags),
    });
    renderPosts();
}
```
